61 template <typename T>
62 inline T mat33_det( const T src[3][3] )
63 {
64  return src[0][0] * (src[1][1] * src[2][2] - src[1][2] * src[2][1])
65  - src[0][1] * (src[1][0] * src[2][2] - src[1][2] * src[2][0])
66  + src[0][2] * (src[1][0] * src[2][1] - src[1][1] * src[2][0]);
67 }

88 template<typename T>
89 inline void mat33_inverse( T dst[3][3], const T src[3][3] )
90 {
91  T inv[3][3];
92 
93  // invert
94  const T det = mat33_det(src);
95 
96  inv[0][0] = + (src[1][1] * src[2][2] - src[1][2] * src[2][1]);
97  inv[0][1] = - (src[0][1] * src[2][2] - src[0][2] * src[2][1]);
98  inv[0][2] = + (src[0][1] * src[1][2] - src[0][2] * src[1][1]);
99  inv[1][0] = - (src[1][0] * src[2][2] - src[1][2] * src[2][0]);
100  inv[1][1] = + (src[0][0] * src[2][2] - src[0][2] * src[2][0]);
101  inv[1][2] = - (src[0][0] * src[1][2] - src[0][2] * src[1][0]);
102  inv[2][0] = + (src[1][0] * src[2][1] - src[1][1] * src[2][0]);
103  inv[2][1] = - (src[0][0] * src[2][1] - src[0][1] * src[2][0]);
104  inv[2][2] = + (src[0][0] * src[1][1] - src[0][1] * src[1][0]);
105 
106  // scale by determinant
107  for( uint32_t i=0; i < 3; i++ )
108  for( uint32_t k=0; k < 3; k++ )
109  dst[i][k] = inv[i][k] / det;
110 }

162 template<typename T>
163 inline int mat33_rank( const T src[3][3] )
164 {
165  T mat[3][3];
166 
167  // reducing to row-echelon form alters the matrix
168  mat33_copy(mat, src);
169 
170  // iteratively compute the rank
171  int rank = 3; // (= #columns)
172 
173  for( int row=0; row < rank; row++ )
174  {
175  if( mat[row][row] != 0 )
176  {
177  for( int col=0; col < 3; col++ )
178  {
179  if (col != row)
180  {
181  const T mult = mat[col][row] / mat[row][row];
182 
183  for( int i = 0; i < rank; i++ )
184  mat[col][i] -= mult * mat[row][i];
185  }
186  }
187  }
188  else
189  {
190  bool reduce = true;
191 
192  for( int i=row+1; i < 3; i++ )
193  {
194  if( mat[i][row] != 0 )
195  {
196  for( int k=0; k < rank; k++ )
197  {
198  const T tmp = mat[row][k];
199  mat[row][k] = mat[i][k];
200  mat[i][k] = tmp;
201  }
202 
203  reduce = false;
204  break ;
205  }
206  }
207 
208  if( reduce )
209  {
210  rank--;
211 
212  for( int i=0; i < 3; i++ )
213  mat[i][row] = mat[i][rank];
214  }
215 
216  row--;
217  }
218  }
219 
220  return rank;
221 }

420  inline void PROFILER_BEGIN( profilerQuery query )
421  {
422  const uint32_t evt = query*2;
423  const uint32_t flag = (1 << query);
424 
425  CUDA(cudaEventRecord(mEventsGPU[evt], mStream));
426  timestamp(&mEventsCPU[evt]);
427 
428  mProfilerQueriesUsed |= flag;
429  mProfilerQueriesDone &= ~flag;
430  }
431 
435  inline void PROFILER_END( profilerQuery query )
436  {
437  const uint32_t evt = query*2+1;
438 
439  CUDA(cudaEventRecord(mEventsGPU[evt]));
440  timestamp(&mEventsCPU[evt]);
441  timespec cpuTime;
442  timeDiff(mEventsCPU[evt-1], mEventsCPU[evt], &cpuTime);
443  mProfilerTimes[query].x = timeFloat(cpuTime);
444 
445  if( mEnableProfiler && query == PROFILER_NETWORK )
446  {
447  printf(LOG_TRT "layer network time - %f ms\n", gProfiler.timingAccumulator);
448  gProfiler.timingAccumulator = 0.0f;
449  printf(LOG_TRT "note -- when processing a single image, run 'sudo jetson_clocks' before\n"
450  " to disable DVFS for more accurate profiling/timing measurements\n");
451  }
452  }
453 
457  inline bool PROFILER_QUERY( profilerQuery query )
458  {
459  const uint32_t flag = (1 << query);
460 
461  if( query == PROFILER_TOTAL )
462  {
463  mProfilerTimes[PROFILER_TOTAL].x = 0.0f;
464  mProfilerTimes[PROFILER_TOTAL].y = 0.0f;
465 
466  for( uint32_t n=0; n < PROFILER_TOTAL; n++ )
467  {
468  if( PROFILER_QUERY((profilerQuery)n) )
469  {
470  mProfilerTimes[PROFILER_TOTAL].x += mProfilerTimes[n].x;
471  mProfilerTimes[PROFILER_TOTAL].y += mProfilerTimes[n].y;
472  }
473  }
474 
475  return true;
476  }
477  else if( mProfilerQueriesUsed & flag )
478  {
479  if( !(mProfilerQueriesDone & flag) )
480  {
481  const uint32_t evt = query*2;
482  float cuda_time = 0.0f;
483  CUDA(cudaEventElapsedTime(&cuda_time, mEventsGPU[evt], mEventsGPU[evt+1]));
484  mProfilerTimes[query].y = cuda_time;
485  mProfilerQueriesDone |= flag;
486  //mProfilerQueriesUsed &= ~flag;
487  }
488 
489  return true;
490  }
491 
492  return false;
493  }

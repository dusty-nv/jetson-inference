622  inline void PROFILER_BEGIN( profilerQuery query )
623  {
624  const uint32_t evt = query*2;
625  const uint32_t flag = (1 << query);
626 
627  CUDA(cudaEventRecord(mEventsGPU[evt], mStream));
628  timestamp(&mEventsCPU[evt]);
629 
630  mProfilerQueriesUsed |= flag;
631  mProfilerQueriesDone &= ~flag;
632  }
633 
637  inline void PROFILER_END( profilerQuery query )
638  {
639  const uint32_t evt = query*2+1;
640 
641  CUDA(cudaEventRecord(mEventsGPU[evt]));
642  timestamp(&mEventsCPU[evt]);
643  timespec cpuTime;
644  timeDiff(mEventsCPU[evt-1], mEventsCPU[evt], &cpuTime);
645  mProfilerTimes[query].x = timeFloat(cpuTime);
646 
647  if( mEnableProfiler && query == PROFILER_NETWORK )
648  {
649  LogVerbose(LOG_TRT "layer network time - %f ms\n", gProfiler.timingAccumulator);
650  gProfiler.timingAccumulator = 0.0f;
651  LogWarning(LOG_TRT "note -- when processing a single image, run 'sudo jetson_clocks' before\n"
652  " to disable DVFS for more accurate profiling/timing measurements\n");
653  }
654  }
655 
659  inline bool PROFILER_QUERY( profilerQuery query )
660  {
661  const uint32_t flag = (1 << query);
662 
663  if( query == PROFILER_TOTAL )
664  {
665  mProfilerTimes[PROFILER_TOTAL].x = 0.0f;
666  mProfilerTimes[PROFILER_TOTAL].y = 0.0f;
667 
668  for( uint32_t n=0; n < PROFILER_TOTAL; n++ )
669  {
670  if( PROFILER_QUERY((profilerQuery)n) )
671  {
672  mProfilerTimes[PROFILER_TOTAL].x += mProfilerTimes[n].x;
673  mProfilerTimes[PROFILER_TOTAL].y += mProfilerTimes[n].y;
674  }
675  }
676 
677  return true;
678  }
679  else if( mProfilerQueriesUsed & flag )
680  {
681  if( !(mProfilerQueriesDone & flag) )
682  {
683  const uint32_t evt = query*2;
684  float cuda_time = 0.0f;
685  CUDA(cudaEventElapsedTime(&cuda_time, mEventsGPU[evt], mEventsGPU[evt+1]));
686  mProfilerTimes[query].y = cuda_time;
687  mProfilerQueriesDone |= flag;
688  //mProfilerQueriesUsed &= ~flag;
689  }
690 
691  return true;
692  }
693 
694  return false;
695  }
